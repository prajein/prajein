### ✱ builder philosophy

```typescript
interface Builder {
  code: "clean";
  architecture: "modular";
  ship: "often";
}

function loop(idea: Idea) {
  while (alive) {
    if (idea.is_complex()) {
      refactor(idea, "clarity > cleverness");
    }
    
    // meaningful > hype
    const product = build(idea, { minimal: true });
    
    // ideas are cheap, execution is everything
    if (product.solves_real_problem()) {
      ship(product);
    }
  }
}
```
